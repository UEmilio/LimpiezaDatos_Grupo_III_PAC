# Evaluación de Métodos de Imputación para Datos Categóricos
# 1. Imputación por el Grupo
# Este código se usa para imputar valores faltantes en la columna 'country' de manera más precisa, basándose en la categoría de 'type'.
# En lugar de asignar un solo valor global para todos los datos faltantes, asigna el valor más frecuente dentro de cada grupo.

# Crea una copia del DataFrame 'data' y la almacena en 'data_grupo' para evitar modificar los datos originales.
data_grupo=data.copy()

# Aplica una función lambda que reemplaza los valores NaN en 'country' con el modo (valor más frecuente) dentro de cada grupo.
data_grupo['country'] = data_grupo.groupby('type')['country'].transform(lambda x: x.fillna(x.mode()[0]))

# 2. Imputación por modelo predictivo
# Utiliza KNN para rellenar valores faltantes en 'country' basándose en datos similares.
# KNN predice los valores en función de patrones en los datos.
# KNN convierte categorías a números y luego reconvierte los valores imputados a su estado original.

# Crea una copia del DataFrame 'data' y la almacena en 'data_mod_pred' para evitar modificar los datos originales.
data_mod_pred=data.copy()

# Importa NumPy, que se usará para manejar valores NaN.
import numpy as np

# Importa el KNNImputer, un método basado en K-Nearest Neighbors para imputar valores faltantes.
from sklearn.impute import KNNImputer

# Importa el LabelEncoder, que convierte valores categóricos en números enteros.
from sklearn.preprocessing import LabelEncoder

# Convertimos la variable categórica a valores numéricos
le = LabelEncoder()
data_mod_pred['country_num'] = le.fit_transform(data_mod_pred['country'].astype(str))  # Convertimos los valores existentes

# Reemplazamos los NaN con un número arbitrario (-1000) para diferenciarlos
data_mod_pred.loc[data_mod_pred['country'].isna(), 'country_num'] = np.nan

# Aplicar KNN Imputer solo a la columna categórica numérica
imputer = KNNImputer(n_neighbors=3, weights="uniform")
data_mod_pred[['country_num']] = imputer.fit_transform(data_mod_pred[['country_num']])

# Redondeamos los valores imputados y los convertimos de nuevo a enteros
data_mod_pred['country_num'] = data_mod_pred['country_num'].round().astype(int)

# Convertimos los valores imputados de nuevo a categorías
data_mod_pred['country'] = le.inverse_transform(data_mod_pred['country_num'])

# Eliminamos columna auxiliar: country_num
data_mod_pred=data_mod_pred.drop(columns=['country_num'])

# 2. Por Entropía (Calcular la entropía de la distribución)
# Este código calcula la entropía de la distribución de la columna 'country' en dos versiones del conjunto de datos:
# - 'data_grupo' → Con valores imputados, del método: Imputación por el Grupo.
# - 'data_mod_pred' → Con valores imputados, del método: Imputación por Modelo Predictivo.

# Se define una función para calcular la entropía de la columna 'country', recibiendo como parámetro un DataFrame.
def calculate_entropy(data):
    counts = data['country'].value_counts(normalize=True)
    return -np.sum(counts * np.log2(counts))

# Calcula y muestra la entropía de los dataset.
print(f"Entropía después de imputación por el Grupo: {calculate_entropy(data_grupo)}")
print(f"Entropía después de imputación por Modelo Predictivo: {calculate_entropy(data_mod_pred)}")
