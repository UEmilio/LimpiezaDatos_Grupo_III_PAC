# Evaluacin de Mtodos de Imputacin para Datos Categricos
# 1. Imputacin por el Grupo
# Este cdigo se usa para imputar valores faltantes en la columna 'country' de manera ms precisa, basndose en la categora de 'type'.
# En lugar de asignar un solo valor global para todos los datos faltantes, asigna el valor ms frecuente dentro de cada grupo.

# Crea una copia del DataFrame 'data' y la almacena en 'data_grupo' para evitar modificar los datos originales.
data_grupo=data.copy()

# Aplica una funcin lambda que reemplaza los valores NaN en 'country' con el modo (valor ms frecuente) dentro de cada grupo.
data_grupo['country'] = data_grupo.groupby('type')['country'].transform(lambda x: x.fillna(x.mode()[0]))

# 2. Imputacin por modelo predictivo
# Utiliza KNN para rellenar valores faltantes en 'country' basndose en datos similares.
# KNN predice los valores en funcin de patrones en los datos.
# KNN convierte categoras a nmeros y luego reconvierte los valores imputados a su estado original.

# Crea una copia del DataFrame 'data' y la almacena en 'data_mod_pred' para evitar modificar los datos originales.
data_mod_pred=data.copy()

# Importa NumPy, que se usar para manejar valores NaN.
import numpy as np

# Importa el KNNImputer, un mtodo basado en K-Nearest Neighbors para imputar valores faltantes.
from sklearn.impute import KNNImputer

# Importa el LabelEncoder, que convierte valores categricos en nmeros enteros.
from sklearn.preprocessing import LabelEncoder

# Convertimos la variable categrica a valores numricos
le = LabelEncoder()
data_mod_pred['country_num'] = le.fit_transform(data_mod_pred['country'].astype(str))  # Convertimos los valores existentes

# Reemplazamos los NaN con un nmero arbitrario (-1000) para diferenciarlos
data_mod_pred.loc[data_mod_pred['country'].isna(), 'country_num'] = np.nan

# Aplicar KNN Imputer solo a la columna categrica numrica
imputer = KNNImputer(n_neighbors=3, weights="uniform")
data_mod_pred[['country_num']] = imputer.fit_transform(data_mod_pred[['country_num']])

# Redondeamos los valores imputados y los convertimos de nuevo a enteros
data_mod_pred['country_num'] = data_mod_pred['country_num'].round().astype(int)

# Convertimos los valores imputados de nuevo a categoras
data_mod_pred['country'] = le.inverse_transform(data_mod_pred['country_num'])

# Eliminamos columna auxiliar: country_num
data_mod_pred=data_mod_pred.drop(columns=['country_num'])

# 2. Por Entropa (Calcular la entropa de la distribucin)
# Este cdigo calcula la entropa de la distribucin de la columna 'country' en dos versiones del conjunto de datos:
# - 'data_grupo'  Con valores imputados, del mtodo: Imputacin por el Grupo.
# - 'data_mod_pred'  Con valores imputados, del mtodo: Imputacin por Modelo Predictivo.

# Se define una funcin para calcular la entropa de la columna 'country', recibiendo como parmetro un DataFrame.
def calculate_entropy(data):
    counts = data['country'].value_counts(normalize=True)
    return -np.sum(counts * np.log2(counts))

# Calcula y muestra la entropa de los dataset.
print(f"Entropa despus de imputacin por el Grupo: {calculate_entropy(data_grupo)}")
